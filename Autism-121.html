def remove_pattern(input_txt, pattern):
    r = re.findall(pattern, input_txt)
    for i in r:
        input_txt = re.sub(i, '', input_txt)
        
    return input_txt

data['tidy_tweet'] = np.vectorize(remove_pattern)(data['text'], "@[\w]*")
